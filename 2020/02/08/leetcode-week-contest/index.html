5340. 统计有序矩阵中的负数

题解

  • 直接统计即可

代码

class Solution {
public:
    int countNegatives(vector>& grid) {
        int res = 0;
        for(int i = 0; i < grid.size(); i++)
            for(int j = 0; j < grid[i].size(); j++)
            {
                if(grid[i][j] < 0)
                {
                    res += grid[i].size() - j;
                    break;
                }
            }
        return res;
    }
};

    }

        vs.push_back(num);

        int res = 1;

5342. 最多可以参加的会议数目

题解

  • 首先给所有会议按开始时间从先到后排序
  • 统计天数最大值$maxd$
  • 将日期从1遍历到$maxd$, 将符合条件的区间都丢入堆，取出会议结束时间最早的会议进行匹配
  • 累加会议数

代码

class Solution {
public:
    int maxEvents(vector>& events) {
        sort(events.begin(), events.end(), [&](vector x, vector y){
            return x[0] < y[0];
        });
        int res = 0;
        int maxd = 0;
        for(int i = 0; i < events.size(); i++)
          maxd = max(maxd, events[i][1]);
        int j = 0;
        priority_queue, greater > que;
        for(int i = 1; i <= maxd; i++)
            {
                for(; j < events.size() && events[j][0] <= i; j++)
                        que.push(events[j][1]);
                while(!que.empty())
                {
                    int t = que.top();
                    que.pop();
                    if(t >= i) {res++; break;}
                }
            }
        return res;
    }
};

5343. 多次求和构造目标数组

题解

  • 反向模拟，每次找到数列中的最大值进行逆推
  • 利用堆优化查找最大值
  • 注意求和会溢出int型

代码

class Solution {
public:
    bool isPossible(vector& target) {
        long long int sum = 0;
        priority_queue q;
        for(int i = 0; i < target.size(); i++)
        {
            sum += target[i];
            q.push(target[i]);
        }

        while(!q.empty())
        {
            int t = q.top();
            if(t == 1) return true;
            q.pop();
            long long int p = sum - t;
            sum = t;
            if(t - p < 1) return false;
            if(t-p != 1) q.push(t-p);
        }
        return true;
    }
};

题解

  • 遍历记录各个数是否出现
  • 遍历查询其倍数是否出现
  • 标记0的个数，特判
  • 由于有负数，并且有double的可能，所以整体前移$ 2 * 10^3 $

代码

class Solution {
public:
    bool checkIfExist(vector& arr) {
        int flag = 0;
        bool exit[10010];
        memset(exit, false, sizeof(exit));
        for(int i = 0; i < arr.size(); i++)
            {
                exit[arr[i] + 2000] = true;
                if(arr[i] == 0) flag ++;
            }
        for(int i = 0; i < arr.size(); i++)
            if(exit[arr[i] * 2 + 2000] && arr[i] != 0) return true;
        if(flag >= 2) return true;
        return false;
    }
};

1347. 制造字母异位词的最小步骤数

题解

  • 按字母统计
  • 累计差额

代码

class Solution {
public:
    int minSteps(string s, string t) {
        int cs[26], ct[26];
        memset(cs, 0, sizeof(cs));
        memset(ct, 0, sizeof(ct));
        for(int i = 0; i < s.size(); i++)
            cs[s[i]-'a'] ++;
        for(int i = 0; i < t.size(); i++)
            ct[t[i]-'a'] ++;
        int res = 0;
        for(int i= 0; i < 26; i++)
            if(ct[i] < cs[i]) res += (cs[i] - ct[i]);
        return res;
    }
};

1342. 将数字变成 0 的操作次数

题解

  • 统计num二进制位数和二进制位上1的个数，答案为其和-1，因为所有除1外所有单数会要减1和右移1位，而1⃣️只要进行两者操作中的一个。
  • 特判num为0

代码

class Solution {
public:
    int numberOfSteps (int num) {
        int res = 0;
        while(num)
        {
            if(num & 1 != 0) res ++;
            res ++;
            num >>= 1;
        }
        return max(0, res-1);
    }
};

1343. 大小为 K 且平均值大于等于阈值的子数组数目

题解

  • 先统计前k个数组元素之和
  • 利用滑动窗口思想，向后滑动，减去第一个元素，加上后一个元素得到下一个子数组的和
  • 判断sum是否满足题目要求

代码

class Solution {
public:
    int numOfSubarrays(vector& arr, int k, int threshold) {
        int sum = 0, res = 0;
        for(int i = 0; i < k; i++)
            sum += arr[i];
        if(sum >= threshold*k) res ++;
        for(int i = k; i < arr.size(); i++)
            {
                sum += (arr[i]-arr[i-k]);
                if(sum >= threshold*k) res ++;
            }
        return res;
    }
};

1345. 跳跃游戏 IV

题解

  • 将所有value相同的元素的下标统计在一起
  • 利用$unordered\_map$构造$ = <数组值:数组下标集合>的键值对$
  • $unordered\_ map$的是基于$hash$实现无序映射，其查找和删除的复杂度大部分情况下为$O(1)$，优于$map$的$O(logn)$, 但相较于$map$，其空间占用更多。
  • 利用$bfs$进行最短距离搜索，由于所有同值的下标只会访问一次，所以可以在扩展后直接删除。
  • 时间复杂度为$O(n)$

代码

class Solution {
public:
    int minJumps(vector& arr) {
      unordered_map> mp;
      for(int i = 0; i < arr.size(); i++)
          mp[arr[i]].push_back(i);
      int* dis = new int[arr.size()];
      for(int i = 0; i < arr.size(); i++)
          dis[i] = -1;
      queue q;
      q.push(0);
      dis[0] = 0;
      while(!q.empty())
      {
        int t = q.front();
        q.pop();
        if(t == arr.size()-1) return dis[t];
        if(t-1 >= 0) {if(dis[t-1] == -1) {dis[t-1] = dis[t] + 1; q.push(t-1);}}
        if(t+1 < arr.size()) {if(dis[t+1] == -1){dis[t+1] = dis[t] + 1; q.push(t+1);}}
        for(int i = 0; i < mp[arr[t]].size(); i++)
        {
          int tmp = mp[arr[t]][i];
          if(dis[tmp] == -1)
          {
            dis[tmp] = dis[t] + 1;
            q.push(tmp);
          }
        }
        mp.erase(arr[t]);
      }
      return -1;
    }
};

1337. 方阵中战斗力最弱的 K 行

题解

  • 统计战斗力
  • 根据战斗力和行标排序
  • 时间复杂度$O(m*n)$

代码

class Solution {
public:
    vector kWeakestRows(vector>& mat, int k) {
        vector res, ans;
        int* capacity = new int[mat.size()];
        for(int i = 0; i < mat.size(); i++)
        {
            res.push_back(i);
            capacity[i] = 0;
            for(int j = 0; j < mat[i].size() && mat[i][j]; j++)
                capacity[i] ++;
        }
        sort(res.begin(), res.end(), [&](int x, int y){
            if(capacity[x] == capacity[y])
                return x < y;
            return capacity[x] < capacity[y];
            });
        for(int i = 0; i < k; i++)
            ans.push_back(res[i]);
        return ans;
    }
};

1338. 数组大小减半

题解

  • 统计数组元素大小的最大值作为统计元素出现频率数组的大小
  • 统计频率并排序
  • 频数从大到小累加到半或过半则停止

代码

class Solution {
public:
    int minSetSize(vector& arr) {
        int maxl = 0;
        for(int i = 0; i < arr.size(); i++)
            maxl = max(maxl, arr[i]);
        maxl++;
        int* cnt = new int[maxl];
        for(int i = 0; i < maxl; i++) cnt[i] = 0;
        for(int i = 0; i < arr.size(); i++)
            cnt[arr[i]] ++;
        sort(cnt, cnt+maxl);
        int ans = 0, sum = 0;
        for(int i = maxl-1; i >= 0; i--)
            if(cnt[i])
                {
                   ans++;
                   sum += cnt[i];
                   if(sum >= arr.size()/2)
                    break;
                }
        return ans;
    }
};

1339. 分裂二叉树的最大乘积

题解

  • 递归求和，将$node->val$ 替换成所在子树的和
  • 递归求积，统计所有子树和整体断开时的积的最大值 = (整树和-子树和) * 子树和，需要用$long long int$存储
  • 结果取余 $10^9 + 7$

代码

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxProduct(TreeNode* root) {
        sum(root);
        return ans(root, root->val) % int(1e9+7);
    }
    int sum(TreeNode* root)
    {
        if(root == NULL) return 0;
        return root->val = sum(root->left) + sum(root->right) + root->val;
    }
    long long int ans(TreeNode* root, int sum){
        if(root == NULL) return 0;
        long long int res = (1LL * (sum - root->val) * root->val) ;
        return max(res, max(ans(root->left, sum), ans(root->right, sum)));
    }
};

1340. 跳跃游戏 V

题解

  • 按高度从低到高排序,即得按高度排列的$a[i]$
  • 第一维从低到高进行DP，第二维DP时在未排序排列左右方向DP，选取未超过d且视线内高度小的进行DP
  • DP方程:$dp[a[i]] = max(dp[a[i]], dp[j]+1), d <= a[i]-j <= d \&\& arr[a[i]] > arr[j]$
  • 统计答案:$res = max(res, dp[a[i]])$
  • 时间复杂度为: $O(n*d + nlogn)$

代码

class Solution {
public:
    int maxJumps(vector& arr, int d) {
        int* a  = new int[arr.size()];
        int* dp = new int[arr.size()];
        for(int i = 0; i < arr.size(); i++)
            a[i] = i;
        sort(a, a + arr.size(), [&](int x, int y){
            return arr[x] < arr[y];
        });
        int res = 1;
        for(int i = 0; i < arr.size(); i++)
            {
                dp[a[i]] = 1;
                for(int j = a[i]-1; j >= 0 && a[i]-j <= d && arr[a[i]] > arr[j]; j--)
                    dp[a[i]] = max(dp[a[i]], dp[j]+1);
                for(int j = a[i]+1; j < arr.size() && j-a[i] <= d && arr[a[i]] > arr[j]; j++)
                    dp[a[i]] = max(dp[a[i]], dp[j]+1);
                res = max(res, dp[a[i]]);
            }
        return res;
    }
};

1332. 删除回文子序列

题解

  • 此题是删除回文子序列，而非字串，子序列只要求顺序不变，不必相邻
  • 因为只有a，b，所以最多的操作也只有2次，单独挑出所有的a或者所有的b进行删除
  • 当原字符串为空的时候返回0，当原字符串为回文串的时候返回1，其他进行上述操作都返回2

代码

class Solution {
public:
    int removePalindromeSub(string s) {
        if(s.size() == 0) return 0;
        bool flag = true;
        for(int i = 0; i <= s.size()/2; i++)
            if(s[i] != s[s.size()-i-1])
                {
                    flag = false;
                    break;
                }
        return flag ? 1 : 2;
    }
};

1334. 阈值距离内邻居最少的城市

题解

  • 利用floyd算法统计出点与点之间的最短距离
  • 利用distanceThreshold进行筛选出符合题目要求的解
  • 还可以考虑用其他最短路算法，Bellmanford，spfa，dijkstra 求最短距离

代码

class Solution {
public:
    int findTheCity(int n, vector>& edges, int distanceThreshold) {
        int dp[110][110];
        memset(dp, 0x3f3f3f3f, sizeof(dp));
        for(int i = 0; i < edges.size(); i++)
            dp[edges[i][0]][edges[i][1]] = dp[edges[i][1]][edges[i][0]] = edges[i][2];
        //  注意k是第一层循环，dp[k][i][j] 表示前k个点作为插点能达到的最短距离
        for(int k = 0; k < n; k++)
            for(int i = 0; i < n; i++)
                for(int j = 0; j < n; j++)
                    dp[i][j] =  min(dp[i][j], dp[i][k] + dp[k][j]);

        int count[110] = {0};
        for(int i = 0; i < n; i++)
            for(int j = i+1; j < n; j++)
                if(dp[i][j] <= distanceThreshold) count[i]++, count[j]++;

        int minc = 0x3f3f3f3f, res;
        for(int i = 0; i < n; i++)
            if(count[i] <= minc)  res = i, minc = count[i];
        return res;
    }
};

1335. 工作计划的最低难度

题解

  • 区间预处理 + 动态规划， 时间复杂度 $O(d*n^2)$
  • $dp[i][j]$ 表示 前i天完成j项工作的总困难
  • $dp[i][j] = min(dp[i-1][k] + diff[k+1][j]) , diff[i][j] 代表从i到j的任务的最小难度 $

代码

class Solution {
public:
    int minDifficulty(vector& jobDifficulty, int d) {
        if(jobDifficulty.size() < d) return -1;
        int diff[310][310] = {0};
        for(int i = 0; i < jobDifficulty.size(); i++)
            {
                diff[i][i] = jobDifficulty[i];
                for(int j = i+1; j < jobDifficulty.size(); j++)
                    diff[i][j] = max(diff[i][j-1], jobDifficulty[j]);
            }
        int dp[10][310];
        memset(dp, 0x3f3f3f3f, sizeof(dp));
        for(int i = 0; i < d; i++)
            for(int j = i; j < jobDifficulty.size(); j++)
                {
                if(i == 0) dp[i][j] = diff[0][j];
                else
                    for(int k = i-1; k < j; k++)
                        dp[i][j] = min(dp[i][j], dp[i-1][k] + diff[k+1][j]);
                }
        int res = dp[d-1][jobDifficulty.size()-1];
        return res == 0x3f3f3f3f ? -1 : res;
    }
};

1323. 6 和 9 组成的最大数字

题解

  • 从后向前扫描，将最前面的一个6变成9
  • 假设并更正，记录差值，详见代码

代码

class Solution {
public:
    int maximum69Number (int num) {
        int res = 0, p = 1, dt = 0;
        while(num)
        {
            if(num%10 == 9) res += p*(num%10);
            else {
                res += (p*9 - dt);
                dt = p*3;
                }
            num /= 10;
            p *= 10;
        }
        return res;
    }
};

1324. 竖直打印单词

题解

  • 将各个单词的起点下标统计记录
  • 统计最大单词长，即竖直打印的行数
  • 最后记得删除行末空格

代码

class Solution {
public:
    vector printVertically(string s) {
        bool flag = true;
    vector start;
    for(int i = 0; i < s.size(); i++)
    {
        if(s[i] != ' ' && flag)
        {
            start.push_back(i);
            flag = false;
        }
        if(s[i] == ' ') flag = true;
    }
    int maxlen = 0;
    int ls = int(s.size());
    for(int i = 0; i < start.size(); i++)
        if(i == start.size() - 1)
            maxlen = max(maxlen, ls-start[i]);
        else maxlen = max(maxlen, start[i+1] - start[i] - 1);

    vector ans;
    for(int i = 0; i < maxlen; i++)
        ans.push_back("");

    int x = 0;
    for(int i = 0; i < maxlen; i++)
        for(int j = 0; j < start.size(); j++)
           if((j == start.size()-1 && start[j] + i < s.size()) || (j < start.size() - 1  && start[j] + i < start[j+1]))
                ans[i] += s[start[j] + i];
           else
                ans[i] += ' ';
    for(int i = 0; i < ans.size(); i++)
    {
        for(int j = ans[i].size()-1; j >= 0; j--)
        {
            if(ans[i][j] != ' ')
            {
                ans[i] = ans[i].substr(0,j+1);
                break;
            }
        }
    }
    return ans;
    }
};

1325. 删除给定值的叶子节点

题解

  • 递归操作

代码

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* removeLeafNodes(TreeNode* root, int target) {
        if(root == NULL) return NULL;
        root->left = removeLeafNodes(root->left, target);
        root->right = removeLeafNodes(root->right, target);
        if(root->left == NULL && root->right == NULL && root->val == target)
            return NULL;
        else return root;
    }
};

1326. 灌溉花园的最少水龙头数目

题解

  • 统计区间并按右边界大小进行排序
  • 区间动态规划， $dp[i]$ 为覆盖从0到第i个区间的右边界所需的最小区间个数
  • 动态转移方程：

代码

class Solution {
public:
  int minTaps(int n, vector& ranges) {
      int* a = new int[n+1];
      int* l = new int[n+1];
      int* r = new int[n+1];
      for(int i = 0; i <= n; i++)
          {
              a[i] = i;
              l[i] = i-ranges[i];
              r[i] = i+ranges[i];
          }
      // [&]表示使用了外部变量a/b
      sort(a, a+n+1, [&](int x, int y){return r[x] < r[y];});
      int* dp = new int[n+1];
      int res = 0x3f3f3f3f;
      for(int i = 0; i <= n; i++)
      {
          dp[i] = l[a[i]] <= 0 ? 1 : 0x3f3f3f3f;
          for(int j = i-1; j >= 0 && r[a[j]] >= l[a[i]]; j--)
          {
              dp[i] = min(dp[i], dp[j] + 1);
          }
          if(r[a[i]] >= n) res = min(res, dp[i]);
      }
      if (res >= 0x3f3f3f3f) return -1;
      return res;
  }
};

class Solution {

      {
        if(check(i) && check(n-i))
          {
            res.push_back(i);
            res.push_back(n-i);

          }

      }
      return true;
    }
};

1318. 或运算的最小翻转次数

题解

按位模拟

代码

class Solution {
public:
    int minFlips(int a, int b, int c) {
      int res = 0;
      for(int i = 1; i <= max(max(a, b), c); i <<= 1)
      {
        if((c & i) == 0)
        {
          if((a & i) != 0) res ++;
          if((b & i) != 0) res ++;
        }
        else
          if((a & i) == 0 && (b & i) == 0) res ++;
      }
      // cout << res << endl;
      return res;
    }
};

1319. 连通网络的操作次数

题解

使用并查集抽选必要连接，查询剩余不必要连接是否够转移成必要连接

代码

class Solution {
public:
    int f[10010];
    void init() {for(int i = 0; i < 10010; i++) f[i] = i;}
    int find(int x) {return x == f[x]? x : f[x] = find(f[x]);}
    bool same(int x, int y)  {return find(x) == find(y);}
    void unit(int x, int y)  {f[find(x)] = find(y);}
    int makeConnected(int n, vector>& connections) {
        int res = n-1;
        if(connections.size() < n-1) return -1;
        init();
        for(int i = 0; i < connections.size(); i++)
        {
            int x = connections[i][0];
            int y = connections[i][1];
            if(!same(x, y))
            {
                res--;
                unit(x, y);
            }   
        }
        return res;
    }
};

1320. 二指输入的的最小距离

题解

  • 二维动态规划问题
  • $dp[i][j]$ 代表第一个手指移动到i，第二个手指移动到j时的最小移动距离。
  • 注意边界条件和初始化动态数组，为从0到任何一个位置移动距离为0。
  • 动态转移, 注意当i，j处于同一位置和相邻位置时可以选一个手指移动范围大于一个字母。

    代码

class Solution {
public:
    int dp[305][305];
    int dis(string &word, int x, int y)
    {
    if(x == 0 || y == 0) return 0;
    x --, y --;
    int dr = ((word[x] - 'A') / 6) - ((word[y] - 'A') / 6);
    int dc = ((word[x] - 'A') % 6) - ((word[y] - 'A') % 6);
    return  abs(dr) + abs(dc);
    }

    int minimumDistance(string word) {
    int maxi = 0x3f3f3f;
    memset(dp, maxi, sizeof(dp));
    dp[0][0] = 0;
    for(int i = 0; i <= word.size(); i++)
    for(int j = 0; j <= word.size(); j++)
      {
        if (i < j)
        dp[i][j] = dp[i][j-1] + dis(word, j-1,j);
        if (i > j)
        dp[i][j] = dp[i-1][j] + dis(word, i-1, i);
        if (abs(i-j) <= 1)
        {
          for(int k = 0; k < i; k++)
            dp[i][j] = min(dp[i][j], dp[k][j] + dis(word, k, i));
          for(int k = 0; k < j; k++)
            dp[i][j] = min(dp[i][j], dp[i][k] + dis(word, k, j));
        }
      }
  int ans = maxi;
  for(int i = 0; i <= word.size(); i++)
    ans = min(ans, min(dp[i][word.size()] , dp[word.size()][i]));
   return ans;
    }
};
