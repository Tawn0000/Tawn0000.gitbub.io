class Solution {
public:
    int maxDepth(string s) {
        int ans = 0, cnt = 0;
        for(int i = 0; i < s.size(); i++)
        {
            if(s[i] == '(') cnt ++;
            if(s[i] == ')') cnt --;

        return ans;
    }
};

class Solution {
public:
    const int maxn = 100 + 10;
    int maximalNetworkRank(int n, vector>& roads) {
        int mp[maxn][maxn], d[maxn], ans = 0;
        memset(mp, 0, sizeof(mp));
        memset(d, 0, sizeof(d));
        for(int i = 0; i < roads.size(); i++)
        {
            d[roads[i][0]]++;
            d[roads[i][1]]++;
            mp[roads[i][0]][roads[i][1]]++;
            mp[roads[i][1]][roads[i][0]]++;
        }
        for(int i = 0; i < n; i++)
            for(int j = i+1; j < n; j++)
                ans = max(ans, d[i]+d[j] - mp[i][j]);
        return ans;
    }
};

class Solution {
public:
    bool isPossible(vector& target) {
        long long int sum = 0;
        priority_queue q;
        for(int i = 0; i < target.size(); i++)
        {
            sum += target[i];
            q.push(target[i]);
        }

        while(!q.empty())
        {
            int t = q.top();
            if(t == 1) return true;
            q.pop();
            long long int p = sum - t;
            sum = t;
            if(t - p < 1) return false;
            if(t-p != 1) q.push(t-p);
        }
        return true;
    }
};

class Solution {

        int cs[26], ct[26];

        memset(ct, 0, sizeof(ct));

class Solution {
public:
    int dp[305][305];
    int dis(string &word, int x, int y)
    {
    if(x == 0 || y == 0) return 0;
    x --, y --;
    int dr = ((word[x] - 'A') / 6) - ((word[y] - 'A') / 6);
    int dc = ((word[x] - 'A') % 6) - ((word[y] - 'A') % 6);
    return  abs(dr) + abs(dc);
    }

    int minimumDistance(string word) {
    int maxi = 0x3f3f3f;
    memset(dp, maxi, sizeof(dp));
    dp[0][0] = 0;
    for(int i = 0; i <= word.size(); i++)
    for(int j = 0; j <= word.size(); j++)
      {
        if (i < j)
        dp[i][j] = dp[i][j-1] + dis(word, j-1,j);
        if (i > j)
        dp[i][j] = dp[i-1][j] + dis(word, i-1, i);
        if (abs(i-j) <= 1)
        {
          for(int k = 0; k < i; k++)
            dp[i][j] = min(dp[i][j], dp[k][j] + dis(word, k, i));
          for(int k = 0; k < j; k++)
            dp[i][j] = min(dp[i][j], dp[i][k] + dis(word, k, j));
        }
      }
  int ans = maxi;
  for(int i = 0; i <= word.size(); i++)
    ans = min(ans, min(dp[i][word.size()] , dp[word.size()][i]));
   return ans;
    }
};
