Captcha Identification through Baidu OCR API (IMG, GIF)

文章字数:   941
阅读时长:   4 分
阅读次数:  
  • 在做爬虫的时候，当访问量累计到一定量或者访问速度的过快的时候，网站会出现验证码的反爬措施。
  • 此时可以手动输入验证码，也可以利用算法来识别自动验证
  • 本文调用的是免费的API只能识别简单的由数字或者字母组成的验证码，可以购买百度的服务或者自己训练模型来满足自己不同的需求。

一、注册

  1. 点击-> 官网 注册一个账号，然后登陆。
  2. 点击右上角控制管理台，找到文字识别，进入，然后新建一个应用。
  3. 获得应用的 AppID API Key Secret Key。

二、说明

  1. 对于单纯的一张验证码图片,,先通过一些简单的图像处理方法处理，如二值化，开运算，然后调用API。

  2. 对于GIP动图验证码，先将gif拆帧，变成多张图片，再对每张图片分别识别，投票决定最后的结果。

  3. 具体实现请看代码

三、代码

import cv2
import requests
import string
import numpy as np
from io import BytesIO
from PIL import Image
from aip import AipOcr
from skimage import measure, morphology

class CaptchaIdentify(object):

    def __init__(self):
        # 填写自己的应用的 AppID API Key Secret Key
        APP_ID = 'xxx'
        API_KEY = 'xxx'
        SECRET_KEY = 'xxx'
        self.client = AipOcr(APP_ID, API_KEY, SECRET_KEY)

    # 识别图片
    def recognize_img(self, img):
        img = self.clarify(img)
        img = self.clear(img)
        #baidu ocr
        options = {}
        options["language_type"] = "ENG"
        success, encoded_image = cv2.imencode('.png', img)
        img = encoded_image.tobytes()
        msg = self.client.basicGeneral(img, options)
        result = msg.get('words_result')    
        if(len(result) > 0):
            temp = result[0].get('words')
            res = ''
            for c in temp:
                if (c >= '0' and c <= '9') or (c >= 'A' and c <= 'Z') or (c >= 'a' and c <= 'z'):
                    res += c
            return res
        return ''

    # 识别动图
    def recognize_gif(self, gif):
        vote = ["", "", "", ""]
        res_img = None
        for i, frame in enumerate(self.iter_frames(gif)):
            res_img = frame
            img = np.array(frame)
            res = self.recognize_img(img)
            if len(res) == 4:
                for i in range(4):
                    vote[i] += res[i]

        res = ''
        for i in range(4):
            if vote[i]:
                res += max(string.digits + string.ascii_uppercase, key=vote[i].count)
        return res, res_img

    # gif2png
    def iter_frames(self, gif):
        try:
            i= 0
            while 1:
                gif.seek(i)
                imframe = gif.copy()
                if i == 0:
                    palette = imframe.getpalette()
                else:
                    imframe.putpalette(palette)
                yield imframe
                i += 1
        except EOFError:
            pass

    # 初步处理图片
    def clarify(self, src):
        gray = cv2.cvtColor(src, cv2.COLOR_GRAY2BGR)  # 二值化
        gray = cv2.cvtColor(gray, cv2.COLOR_BGR2GRAY)  # 二值化
        ret, binary = cv2.threshold(src, 0, 255, cv2.THRESH_BINARY_INV | cv2.THRESH_OTSU)  # 过滤浅色
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (1, 2))
        binary = cv2.morphologyEx(binary, cv2.MORPH_OPEN, kernel)   # 开运算
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (1, 3))
        binary = cv2.morphologyEx(binary, cv2.MORPH_OPEN, kernel)
        cv2.bitwise_not(binary, binary)
        return binary

    # 一个简单处理干扰线的方法
    def clear(self, img):
        fx = [0, 0, 0, 1, -1]
        fy = [0, 1, -1, 0, 0]
        h = img.shape[0]
        w = img.shape[1]
        data = np.zeros((h,w))
        for i in range(h):
            for j in range(w):
                count = 1
                num = 0
                for k in range(5):
                    ix = i + fx[k]
                    iy = j + fy[k]
                    if ix >= 0 and ix < h and iy >= 0 and iy < w:
                        count += 1
                        if img[ix][iy] == 0:
                            num += 1
                if num <= 3:
                    data[i][j] = 255
        return data

    # 下载动图
    def download(self, l, r):
        '''
        https://stackoverflow.com/questions/39534830/how-to-download-this-gifdynamic-by-python
        '''
        url = 'xxx' # 图片链接
        for i in range(1, r):
            with open('captcha/%d.gif', 'wb') as f:
                f.write(requests.get(url).content)

    def test_img(self):
        captcha = CaptchaIdentify()
        img = Image.open('img/stdoutput1.png')
        img = np.array(img)
        ans = captcha.recognize_img(img)
        print(ans)

    def test_gif(self):
        captcha = CaptchaIdentify()
        plt.tight_layout(pad=1.08, h_pad=1.5, w_pad=None, rect=None)
        plt.axis('off')
        for i in range(50):
            gif = Image.open('captcha/%d.gif' % i)
            ax = plt.subplot(25, 2, i+1)
            ans, img = captcha.recognize_gif(gif)
            ax.set_title(ans)
            plt.imshow(img)
            print('%2d:' % i + ans)

    def recognize_gif_url(self, url):
        url = 'xxx'# 图片链接
        res, img = captcha.recognize_gif(Image.open(BytesIO(requests.get(url).content)))
#         plt.imshow(img)
#         plt.axis('off')
        return res
