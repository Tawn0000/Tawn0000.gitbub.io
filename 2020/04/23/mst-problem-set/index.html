1. POJ 3723 Conscription

【题目描述】

温迪有一个国家，他想建立一支军队来保护他的国家。他收留了N个女孩和M个男孩，想把她们收留成他的士兵。征兵无特权，必须交纳一万元。女孩和男孩之间有一些关系，温迪可以利用这些关系来降低他的成本。如果X女孩和Y男孩有D关系，并且其中一个已经被收集，Windy可以用10000-D人民币收集另一个。现在考虑到男孩和女孩之间的所有关系，你的任务是找到风必须支付的最少的钱。注意，收集一个士兵时只能使用一个关系。

【输入】

输入的第一行是测试用例的数量。 每个测试用例的第一行包含三个整数，n、m和r。 然后R行，每个包含三个整数Xi，Yi和DI。 每个测试用例前都有一个空白行。
1 ≤ N, M ≤ 10000
0 ≤ R ≤ 50,000
0 ≤ xi < N
0 ≤ yi < M
0 < di < 10000

【输出】

对于每个测试样例输出答案占单独一行

【样例输入】
2

5 5 8
4 3 6831
1 3 4583
0 0 6592
0 1 3063
3 3 4975
1 3 2049
4 2 2104
2 2 781

5 5 10
2 4 9820
3 2 6236
3 1 8864
2 4 8326
2 0 5156
2 0 1463
4 1 2439
0 4 4373
3 4 8889
2 4 3133

【样例输出】
71071
54223

【解析】

  • 将所有有关系的男孩和女孩建边，边权为D。
  • 跑最大生成树，ans = (n+m) * 10000-s, s s为最大生成树代价，从这个等式可以看出，s越大，ans越小，这也是为什么要跑最大生成树的原因。
  • 最大生成树： 最小生成树（改变排序规则，边权从大到小排序）。
  • 如果没关系也建边，跑MST，会TLE。

【代码】

#include<cstdio>
#include<algorithm>
using namespace std;
const int INF = 0x3f3f3f3f;
const int maxn = 1e5+10;
const int maxm = 1e5+10;
int T,n,m,R;
int f[maxn];
struct edge{
  int u,v,cost;
  edge(int u = 0, int v = 0, int cost = 0) : u(u), v(v), cost(cost){}
}es[maxn];

bool cmp(edge &a, edge &b){
  return a.cost > b.cost;
}

void init()
{
  for(int i = 0; i <= n+m; i++)
      f[i] = i;
}

int find(int x)
{
  return f[x] == x ? x : f[x] = find(f[x]);
}

bool same(int x, int y)
{
  return find(x) == find(y);
}

void unit(int x, int y)
{
  f[find(x)] = find(y);
}

int main()
{
  scanf("%d",&T);
  while(T--)
  {
    int s = 0;
    scanf("%d%d%d",&n, &m, &R);
    init();
    for(int i = 0; i < R; i++)
     {
       int u,v,c;
       scanf("%d%d%d",&u,&v,&c);
       es[i] = edge(u,v+n,c);
     }
     sort(es, es+R, cmp);
     int t = 0;
     for(int i = 0; i < R; i++)
     {
       if(!same(es[i].u,es[i].v))
       {
         unit(es[i].u,es[i].v);
         s += es[i].cost;
         t++;
       }
       if(t == n+m-1) break;
     }
     //cout << s << endl;
     printf("%d\n",(n+m) * 10000-s);
  }
  return 0;
}

2. POJ 2377 Bad Cowtractors

【题目描述】

贝茜受雇在农场主约翰的N个谷仓(2 <= N <= 1000)之间建立一个廉价的互联网网络。FJ已经做了一些调查，发现了M (1 <= M <= 20,000)对谷仓之间可能的连接路径。每个可能的连接路由都有一个相关的成本C (1 <= C <= 100,000)。农民约翰想要在连接网络上花费最少的钱;他甚至都不想付钱给贝茜。意识到农场主约翰不会给她钱，贝茜决定做最坏的工作。她必须决定一组连接安装,所以(我)这些连接的总成本是尽可能大,(2)所有的谷仓都连接在一起(这样就可以达到任何谷仓从其他仓库通过安装连接的路径),和(iii),这样没有周期之间的连接(农民约翰很容易能够检测)。条件(ii)和(iii)确保最后的连接集看起来像一个“树”。

【输入】

第1行:两个空格分隔的整数:N和M*线2 . .M+1:每行包含三个以空格分隔的整数A、B和C，它们描述了成本为C的仓库A和仓库B之间的连接路由。

【输出】

第1行:单个整数，包含连接所有仓库的最昂贵树的价格。如果不可能连接所有的谷仓，输出-1。

【样例输入】
5 8
1 2 3
1 3 7
2 3 10
2 4 4
2 5 8
3 4 6
3 5 2
4 5 17

【样例输出】
42

【提示】

输出详细信息:

最昂贵的树花费17 + 8 + 10 + 7 = 42。它使用以下连接:4到5、2到5、2到3和1到3。

【解析】

  • 题目说明成本最大且连接成树，所以直接跑最大生成树即可，代码和上题类似。

     return a.cost > b.cost;

{
    for (int i = 0; i < E; i++)
    {
        scanf("%d%d%d",&es[i].u,&es[i].v,&es[i].cost);

    for(int i = 0; i < V; i++)
        f[i] = i;
    return ;

    else
        return f[x] = find(f[x]);
}

{
    x = find(x);
    y = find(y);
    if(x == y) return;

    return ;
}

{
    x = find(x);
    y = find(y);

    input();
    sort (es,es+E,comp);

      for(int j = 0; j < E; j++)

4. FZU 2254 英语考试

【题目描述】

在过三个礼拜，YellowStar有一场专业英语考试，因此它必须着手开始复习。

这天，YellowStar准备了n个需要背的单词，每个单词的长度均为m。

YellowSatr准备采用联想记忆法来背诵这n个单词：

1、如果YellowStar凭空背下一个新词T，需要消耗单词长度m的精力

2、如果YellowSatr之前已经背诵了一些单词，它可以选择其中一个单词Si，然后通过联想记忆的方法去背诵新词T，需要消耗的精力为hamming(Si, T) * w。

hamming(Si, T)指的是字符串Si与T的汉明距离，它表示两个等长字符串之间的汉明距离是两个字符串对应位置的不同字符的个数。

由于YellowStar还有大量繁重的行政工作，因此它想消耗最少的精力背诵下这n个单词，请问它最少需要消耗多少精力。

【输入】

包含多组测试数据。

第一行为n, m, w。

接下来n个字符串，每个字符串长度为m，每个单词均为小写字母’a’-‘z’组成。

1≤n≤1000

1≤m, w≤10

【输出】

输出一个值表示答案。

【样例输入】

3 4 2
abch
abcd
efgh

【样例输出】

10

【提示】

最优方案是：先凭空记下abcd和efgh消耗精力8,在通过abcd联想记忆去背诵abch，汉明距离为1,消耗为1 * w = 2,总消耗为10。

【解析】

  • 以汉明距离和m的最小值为边权建边，然后跑最小生成树。
  • 需要注意的是：此图为稠密图，建议使用Prime算法， 朴素版O(n^2)。
  • Kruskal的算法复杂度为：O(n^2logn^2), 需要使用scanf，printf 输入输出数据，否则会TLE

【代码】

  • Prime 朴素做法
//prime 临接矩阵，朴素版，
#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
const int maxn = 1e3 + 10;
const int INF = 0x3f3f3f3f;
int n, m, w;
char s[maxn][11];

int g[maxn][maxn];
int dis[maxn];
bool vis[maxn];

int prime(int s)
{
    memset(dis, INF, sizeof(dis));
    memset(vis, false, sizeof(vis));
    dis[s] = 0;
    int res = 0;
    for(int i = 0; i < n; i++)
    {
        int k=0, mi = INF; //记录最小距离及其标号
        for(int j = 0; j < n; j++)
            if(!vis[j] && mi > dis[j])
                {
                    k = j;
                    mi = dis[j];
                }
        res += dis[k];
        vis[k] = true;
        for(int j = 0; j < n; j++)
            if(!vis[j] && dis[j] > g[k][j])
                dis[j] = g[k][j];
    }
    return res;
}


int hamming(int x, int y)
{
  int res = 0;
  for(int i = 0; i < m; i++)
    res += (s[x][i] != s[y][i]);
  return res;
}

int main()
{
  while(~scanf("%d%d%d", &n, &m, &w))
  {
  for(int i = 0; i < n; i++) scanf("%s", s[i]);
  for(int i = 0; i < n; i++)
    for(int j = i+1; j < n; j++)
        g[i][j] = g[j][i] = min(m, w*hamming(i, j));
  int ans = prime(0) + m;
  printf("%d\n", ans);
}
  return 0;
}

//prime_heap
#include <iostream>
#include <algorithm>
#include <cstdio>
#include <queue>

using namespace std;
typedef pair<int,int> P;
const int maxn = 1e3 + 10;
const int maxm = 1e6 + 10;
const int INF = 0x3f3f3f3f;
int n, m, w;
char s[maxn][11];

int tot = 0;
int h[maxn];//链式前向星

struct edge
{
  int to, c, next;
  edge(int x = 0, int y = 0, int z = 0) : to(x), c(y), next(z) {}
 }es[maxm*2];//记录边,注意是2倍

void add_edge(int u, int v, int c)
{
    es[tot] = edge(v,c,h[u]);
    h[u] = tot++;
    es[tot] = edge(u,c,h[v]);
    h[v] = tot++;
}

int dis[maxn];
bool vis[maxn];

int prime_heap(int s)
{
    int res = 0;
    priority_queue<P, vector<P>, greater<P> > q;
    memset(dis, INF, sizeof(dis));
    memset(vis, false, sizeof(vis));
    dis[s] = 0;
    q.push(P(dis[s], 0));
    while(!q.empty())
    {
        P p = q.top();
        q.pop();
        int u = p.second;
        if(vis[u]) continue;
        res += p.first;
        vis[u] = true;
        for(int i = h[u]; ~i; i=es[i].next)
        {
            int v = es[i].to;
            if(!vis[v] && dis[v] > es[i].c)
            {
                dis[v] = es[i].c;
                q.push(P(dis[v], v));
            }
        }
    }
    return res;
}

int hamming(int x, int y)
{
  int res = 0;
  for(int i = 0; i < m; i++)
    res += (s[x][i] != s[y][i]);
  return res;
}

int main()
{
  while(~scanf("%d%d%d", &n, &m, &w))
  {
      tot = 0;
      memset(h, -1, sizeof(h));
      for(int i = 0; i < n; i++) scanf("%s", s[i]);
      for(int i = 0; i < n; i++)
        for(int j = i+1; j < n; j++)
            add_edge(i,j,min(m, w*hamming(i, j)));
      int ans = prime_heap(0) + m;
      printf("%d\n", ans);
 }
  return 0;
}

5. POJ 1182 食物链

【题目描述】

动物王国中有三类动物A,B,C，这三类动物的食物链构成了有趣的环形。A吃B， B吃C，C吃A。
现有N个动物，以1－N编号。每个动物都是A,B,C中的一种，但是我们并不知道它到底是哪一种。
有人用两种说法对这N个动物所构成的食物链关系进行描述：
第一种说法是”1 X Y”，表示X和Y是同类。
第二种说法是”2 X Y”，表示X吃Y。
此人对N个动物，用上述两种说法，一句接一句地说出K句话，这K句话有的是真的，有的是假的。当一句话满足下列三条之一时，这句话就是假话，否则就是真话。
1） 当前的话与前面的某些真的话冲突，就是假话；
2） 当前的话中X或Y比N大，就是假话；
3） 当前的话表示X吃X，就是假话。
你的任务是根据给定的N（1 <= N <= 50,000）和K句话（0 <= K <= 100,000），输出假话的总数。

【输入】

第一行是两个整数N和K，以一个空格分隔。
以下K行每行是三个正整数 D，X，Y，两数之间用一个空格隔开，其中D表示说法的种类。
若D=1，则表示X和Y是同类。
若D=2，则表示X吃Y。

【输出】

只有一个整数，表示假话的数目。

【样例输入】

100 7
1 101 1
2 1 2
2 2 3
2 3 3
1 1 3
2 3 1
1 5 5

【样例输出】

3

【解析】

  • 设计一种机制，来判定吃与被吃的关系
    • x吃y： x-2y 2x-3y 3x-y
    • y吃x： y-2x 2y-3x 3y-x
    • x同y： x-y 2x-2y 3x-3y
  • 所以要建立一个“滚动并查集”，来记录已知关系
  • 之所以要建立“滚动并查集” 而不是一个并查集是因为我们只能标记吃与被吃的关系，而不能标记其类别关系
    因为我们不知道其具体是A，是B，还是C，标记同类关系并不能判断时候的吃与被吃关系是否正确。

【代码】

#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
const int maxn = 5e4 + 10;
int n,k;
int f[maxn*3];
void init(){ for(int i = 0; i <= n*3+10; i++) f[i] = i;}
int find(int x){return x == f[x] ? f[x] : f[x] = find(f[x]);}
void unit(int x, int y){f[find(x)] = find(y);}
bool same(int x, int y){return find(x) == find(y);}

int main()
{
  scanf("%d%d",&n,&k);
  init();
  int ans = 0;
  for(int i = 0; i < k; i++)
  {
    int d,x,y;
    scanf("%d%d%d",&d,&x,&y);
    if(x > n || y > n) {ans++;continue;}
    if(d == 1){
      if(same(x,y+n) || same(x,y+2*n)) ans++;
      else {unit(x, y); unit(x+n, y+n); unit(x+2*n, y+2*n);}
    }
    else {
      if(same(x, y) || same(x, y+2*n)) ans++;
      else {unit(x, y+n); unit(x+n, y+2*n); unit(x+2*n, y);}
    }
  }
  printf("%d\n",ans);
  return 0;
}

6. POJ 2236 Wireless Network

【题目描述】

东南亚发生地震。ACM(亚洲合作医疗小组)已经与膝上电脑建立了一个无线网络，但是一个意想不到的余震袭击了，网络中的所有电脑都坏了。电脑一个接一个地修好了，网络又渐渐开始工作了。由于硬件的限制，每台计算机只能与距离它不超过d米的计算机直接通信。但是每一台计算机都可以看作是另外两台计算机之间通信的媒介，也就是说，如果计算机A和计算机B可以直接通信，那么计算机A和计算机B就可以通信，或者有一台计算机C可以同时与计算机A和计算机B通信。

在修复网络的过程中，工人可以随时进行两种操作，一种是修复一台计算机，另一种是测试两台计算机是否可以通信。你的工作是回答所有的测试操作。

【输入】

第一行包含两个整数N和d (1 <= N <= 1001, 0 <= d <= 20000)。这里N是计算机的数量，从1到N编号，D是两台计算机可以直接通信的最大距离。在接下来的N行中，每一行包含两个整数xi, yi (0 <= xi, yi <= 10000)，这是N台计算机的坐标。从(N+1)-第一行到输入结束，有一个接一个的操作。每一行包含以下两种格式中的一种操作:

  1. “O p”(1 <= p <= N)，表示修复计算机p。
  2. “S p q”(1 <= p, q <= N)，表示测试计算机p和q是否可以通信。

输入不会超过300000行。

【输出】

对于每个测试操作，如果两台计算机能够通信，则打印“成功”，如果不能通信，则打印“失败”。

【样例输入】

4 1
0 1
0 2
0 3
0 4
O 1
O 2
O 4
S 1 4
O 3
S 1 4

【样例输出】

FAIL
SUCCESS

【解析】

  • 这道题很明显考察并查集，我们需要用并查集维护连通关系，先存点，然后利用坐标求出各自距离D，如果D<=d, 则连通两点，unit操作。
  • 对O操作，使用unit操作连接两个连通集合
  • 对S操作，使用same操作检查两点是否在一个连通集合

【代码】

#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

struct Node {
   int x;
   int y;
};
Node P[1010];
int f[1010];
int n;
double d;

int find(int x) {return f[x] == x ? x : f[x] = find(f[x]);}
void unit(int x, int y) {f[find(x)]=find(y);}

double dis(Node a,Node b){
    return (double)((a.x-b.x) * (a.x-b.x) + (a.y-b.y) * (a.y-b.y));
}
void solve()
{
    memset(P,0,sizeof(P));
    for (int i = 1; i <= n; i++)
        scanf("%d %d",&P[i].x,&P[i].y);
    d*=d;
    int a,b;
    char c;
    while (cin>>c)
    {
        if (c == 'O')
        {
            scanf("%d",&a);
            f[a] = a;
            for (int i = 1; i <= n; i++)
            {
                if ( a == i)  continue;
                if(f[i] != 0&&dis(P[i],P[a])<=d)
                unit(i,a);
            }
        }
        else {
              scanf("%d %d",&a,&b);
             if(find(a) == find(b))
               printf("SUCCESS\n");
             else printf("FAIL\n");
        }
    }
    return ;
}
int main ()
{
   scanf("%d %lf",&n,&d);
   solve();
   return 0;
}

7. Codeforces 891 C Envy

【题目描述】

For a connected undirected weighted graph G, MST (minimum spanning tree) is a subgraph of G that contains all of G’s vertices, is a tree, and sum of its edges is minimum possible.

You are given a graph G. If you run a MST algorithm on graph it would give you only one MST and it causes other edges to become jealous. You are given some queries, each query contains a set of edges of graph G, and you should determine whether there is a MST containing all these edges or not.

【输入】

The first line contains two integers n, m (2  ≤ n, m  ≤ 5·105, n - 1 ≤ m) — the number of vertices and edges in the graph and the number of queries.

The i-th of the next m lines contains three integers u i, v i, w i ( u i ≠ v i, 1 ≤ w i ≤ 5·105) — the endpoints and weight of the i-th edge. There can be more than one edges between two vertices. It’s guaranteed that the given graph is connected.

The next line contains a single integer q (1 ≤ q ≤ 5·105) — the number of queries.

q lines follow, the i-th of them contains the i-th query. It starts with an integer k i (1 ≤ k i ≤ n - 1) — the size of edges subset and continues with k i distinct space-separated integers from 1 to m — the indices of the edges. It is guaranteed that the sum of k i for 1 ≤ i ≤ q does not exceed 5·105.

【输出】

For each query you should print “YES” (without quotes) if there’s a MST containing these edges and “NO” (of course without quotes again) otherwise.

【样例输入】

5 7
1 2 2
1 3 2
2 3 1
2 4 1
3 4 1
3 5 2
4 5 2
4
2 3 4
3 3 4 5
2 1 7
2 1 2

【样例输出】

YES
NO
YES
NO

【题解】

1.kruskal + dsu
2.提前将query 中 e.cost = w 的存下减小复杂度
3.每次利用kruskal贪心边时，若贪心的边的权重变大，则先判断Q[w]中的各个询问是否满足
即无环形成，通过w前的dsu判断，记得复原dsu（reload）
原因：因为w前的最小生成森林虽然不唯一，但是点集唯一，若qi中所有wi的边都可以在wi时放入，则证明存在一个MST包含qi中所有边。

注意：
wrong 可能是 该调用find_()函数是调用了find();
t 可能是没有将query根据w离线来查询;

【代码】

#include<cstdio>
#include<iostream>
#include<cstdlib>
#include<algorithm>
#include<ctime>
#include<cctype>
#include<cmath>
#include<string>
#include<cstring>

#include<queue>
#include<list>
#include<vector>
#include<map>
#include<set>
#include<complex>
#include<iomanip>
#define sqr(x) ((x)*(x))
#define MOD 1e9+7
#define INF 0x3f3f3f3f
const double PI = acos(-1.0);
const double eps = 1e-6;
const int maxm =  5e5 + 100;
const int maxn =  5e5 + 100;
typedef long long LL;
using namespace std;

int n,m,q;
vector<pair <int,int> > Q[maxn];
bool vis[maxn];

int f[maxn];

void init()
{
    for(int i = 1; i <= n; i++)
        f[i] = i;
}

int find(int x)
{
    return  f[x] == x? x :f[x] =  find(f[x]);
}

bool same(int x, int y)
{
    return find(x) == find(y);
}

void unit(int x, int y)
{
    x = find(x);
    y = find(y);
    if(x != y)
        f[x] = y;
}

struct edge{
    int u,v,cost;
};

bool cmp(const edge& e1, const edge& e2)
{
    return e1.cost < e2.cost;
}
edge es[maxm],et[maxm];
int top = 0,val[maxn];
int *s[maxn]; //be used to reload

int find_(int x){
    if(x==f[x]) return x;
    s[++top]=f+x;
    val[top]=f[x];
    f[x]=find_(f[x]);
    return f[x];
}

void reload()
{
    while(top){
        (*s[top])=val[top];--top;
    }
}

void solve(int& w)
{

    int ed = Q[w].size();
    int r = -1;
    for(int j = 0; j < ed; j++)
    {
        pair<int,int> l;
        l = Q[w][j];
        int y = l.first;
        if(!vis[y]) continue;
        int x = l.second;
        if(y > r){reload();r = y;};
        int p1=find_(et[x].u);
        int p2=find_(et[x].v);
        if(p1!=p2){
            s[++top]=f+p1;
            val[top]=f[p1];
            f[p1]=p2;
        }
        else vis[y] = false;
    }
    reload();//don't forget it
}
int main()
{
    scanf("%d%d",&n,&m);
    init();
    for(int i = 1; i <= m; i++)
    {scanf("%d%d%d",&es[i].u,&es[i].v,&es[i].cost);et[i] = es[i];}
    sort(es+1,es+m+1,cmp);
    scanf("%d",&q);
    memset(vis,true,sizeof(vis));
    for(int i = 0; i < q; i++)
    {
        int len;
        scanf("%d",&len);
        for(int j = 0; j < len; j++)
        {
            scanf("%d",&t);
            pair<int,int> pt;
            int w = et[t].cost;
            pt.first = i;
            pt.second = t;
            Q[w].push_back(pt);
        }

    }

    int value = -1;
    for(int i = 1; i <= m; i++)
    {
        edge e = es[i];
        if(e.cost > value)
        {
            value = e.cost;
            solve(value);
        }

        if(!same(e.u,e.v))
            unit(e.u,e.v);
    }

    for(int i = 0; i < q; i++)
        if(vis[i])  printf("YES\n");
        else printf("NO\n");
    return 0;
}

{
    for(int i = 0; i < P.size(); i++)
        f[P[i].first] = P[i].second;

void solve(int& value)
{
    for(int j = 0; j < q; j++)//TLE的原因
    {
        if(!vis[j]) continue;
        P.clear();
        int k  = idx[j];
        for(; k < Q[j].size(); k++)
        {
            int x = Q[j][k].second-1;
            if(et[x].cost == value)

                int fx = find_(et[x].u);

                if(fx == fy)

                p.second = f[fx];

}
