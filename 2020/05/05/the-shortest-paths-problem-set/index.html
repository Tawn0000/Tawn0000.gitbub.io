1. HDU1548 一个奇怪的电梯

【题目描述】

有个奇怪的电梯。电梯可以根据您的需要在每一层进行停留，每一层都有一个数字Ki(0 <= Ki <= N)。电梯只有两个按钮:向上和向下。当你在第i层时，如果你按下“向上”按钮，你就会爬上第i层。同样的，如果你按下“向下”按钮，你就会向下走到i+Ki层，同样的，如果你按下“向下”按钮，你就会向下走到i+Ki层。e，你去i-Ki层。当然，层数不能比N大，也不能比1小。例如，有一栋5层的建筑，k1 = 3, k2 = 3,k3 = 1,k4 = 2, k5 = 5。从1楼开始，你可以按“上”键，然后你就可以上4楼了，如果你按“下”键，电梯就做不到，因为它不能下到-2楼，你知道，-2楼是不存在的。
问题来了:当你在A层，你想去B层，他至少要按几次“上”或“下”键?

【输入】

输入由几个测试用例组成。，每个测试用例包含两行。
第一行包含三个整数N,A,B (1 <= N,A,B <= 200)，描述上述，第二行包含N个整数k1,k2，…kn。
单个0表示输入的结束。

【输出】

对于每一个输入输出接口，当你在A层，你想要去B层时，你需要按下按钮的次数最少。如果你不能到达B层，打印 “-1”。

【样例输入】

5 1 5
3 3 1 2 5
0

【样例输出】

3

【解析】

以层为点，以电梯上下为边建边，边权为1。 所以此题可以用Dijkstra也可以用BFS。注意边不能到达1以下，n以上。

【代码】

// BFS
#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
typedef pair <int,int> P;
const int maxn = 200+10;
int N;
int A,B;
int a[maxn];
bool vis[maxn];

int bfs()
{
    queue <P> Q;
    Q.push(P(A,0));
    memset(vis,false,maxn);
    vis[A] = true;

    while(!Q.empty())
    {
        P t = Q.front();
        Q.pop();
        if(t.first == B)  return t.second;
        vis[t.first] = true;
        int up = t.first + a[t.first];
        int down = t.first - a[t.first];
        if(up <= N && !vis[up])   Q.push(P(up,t.second+1));
        if(down > 0&& !vis[down]) Q.push(P(down,t.second+1));
    }
   return -1;
}

int main()
{
  while(scanf("%d",&N) && N)
  {
      scanf("%d%d",&A,&B);
      for(int i = 1; i <= N; i++)
        scanf("%d",&a[i]);
      printf("%d\n",bfs());
  }
  return 0;
}
//Dijkstra
#include<bits/stdc++.h>

#define INF 0x3f3f3f
const int MAXN = 1e3 + 5;
using namespace std;
typedef pair<int,int> p;
priority_queue< p,vector<p>,greater<p> > q;
int n,a,b,i,d[MAXN],k[MAXN];
int dij(int n,int a,int b){
    while(!q.empty()){
        q.pop();
    }
    memset(d,INF,sizeof(d));
    d[a]=0;
    q.push(p(0,a));
    while(!q.empty()){
        p x=q.top();
        q.pop();
        int dist=x.first,u=x.second;
        if(u==b){
            return d[u];
        }
        if(d[u]<dist) continue;
        int up=u+k[u];
        int down=u-k[u];
    //    cout << u << " " << up << " " << down << endl;
        if(up<=n&&d[up]>d[u]+1){
            d[up]=d[u]+1;
            q.push(p(d[up],up));
        }
        if(down>=1&&d[down]>d[u]+1){
            d[down]=d[u]+1;
            q.push(p(d[down],down));
        }
    }
    return -1;
}
int main(){
    while(1){
        cin>>n;
        if(n==0) break;
        cin>>a>>b;
        for(int i=1;i<=n;i++){
            cin>>k[i];
        }
        cout<<dij(n,a,b)<<endl;
        // for(int i = 1; i <= n; i++)
        //    cout << d[i] << endl;
    //    return 0;
    }
    return 0;
}

2. HDU2544 最短路

【题目描述】

在每年的校赛里，所有进入决赛的同学都会获得一件很漂亮的t-shirt。但是每当我们的工作人员把上百件的衣服从商店运回到赛场的时候，却是非常累的！所以现在他们想要寻找最短的从商店到赛场的路线，你可以帮助他们吗？

【输入】

输入包括多组数据。每组数据第一行是两个整数N、M（N<=100，M<=10000），N表示成都的大街上有几个路口，标号为1的路口是商店所在地，标号为N的路口是赛场所在地，M则表示在成都有几条路。N=M=0表示输入结束。接下来M行，每行包括3个整数A，B，C（1<=A,B<=N,1<=C<=1000）,表示在路口A与路口B之间有一条路，我们的工作人员需要C分钟的时间走过这条路。
输入保证至少存在1条商店到赛场的路线。

【输出】

对于每组输入，输出一行，表示工作人员从商店走到赛场的最短时间

【样例输入】

2 1
1 2 3
3 3
1 2 5
2 3 5
3 1 2
0 0

【样例输出】

3
2

【解析】

模版题，数据也比较水，一个用来练习最短路的模版们

【代码】

//Dijkstra
#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
typedef pair <int,int> P;
const int maxn = 100+10;
const int INF = 0x3f3f3f3f;
int n,m;
int d[maxn];

struct edge
{
    int to,cost;
    edge(int u = 0,int d = 0) : to(u),cost(d) {}
};

vector <edge> G[maxn];

void dijkstra()
{
    memset(d,INF,sizeof(d));
    d[1] = 0;
    priority_queue < P,vector<P>,greater<P> >  Q;//P.first 是距离，.P.second 是顶点
    Q.push(P(0,1));
    while(!Q.empty())
    {
        P x = Q.top();
        Q.pop();
        int dist = x.first;
        int u = x.second;

        if(d[u] < dist) continue;//排除u相同时, dist不同的情况。
        for(int i = 0; i < G[u].size(); i++)
        {
            edge e = G[u][i];
            if(d[e.to] > d[u] + e.cost)
            {
                d[e.to] = d[u]+ e.cost;
                Q.push(P(d[e.to],e.to));
            }
        }
    }
    return ;
}

void input()
{
    for(int i = 0; i <= n; i++ )
        G[i].clear();
    for(int i = 0; i < m; i++)
    {
      int u,v,dist;
      scanf("%d%d%d",&u,&v,&dist);
      G[u].push_back(edge(v,dist));
      G[v].push_back(edge(u,dist));
    }
}

int main()
{
    while(scanf("%d%d",&n,&m) && (n || m))
    {
        input();
        dijkstra();
        printf("%d\n",d[n]);
     }
    return 0;
}
//Spfa
#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
typedef pair <int,int> P;
const int maxn = 100+10;
const int maxm = 1e4 + 100;
const int INF = 0x3f3f3f3f;
int n,m;
int d[maxn];
int cnt[maxn];
bool inq[maxn];
struct edge
{
    int to,cost;
};

vector <edge> G[maxn];

bool Spfa()
{
    memset(d,INF,sizeof(d));
    memset(cnt,0,sizeof(cnt));
    memset(inq,false,sizeof(inq));
    queue <int> Q;
    Q.push(1);
    d[1] = 0;
    cnt[1] = 1;
    inq[1] = true;
    while(!Q.empty())
    {
        int t = Q.front();
        Q.pop();
        inq[t] = false;
        for(int i = 0; i < G[t].size(); i++)
        {
            edge e = G[t][i];
            if(d[e.to] > d[t] + e.cost)
            {
                d[e.to] = d[t] + e.cost;
                if(!inq[e.to]) {
                    if(++cnt[e.to] > n)  return false;
                    Q.push(e.to);
                    inq[e.to] = true;
                }
            }
        }
    }
    return true;
}

int main()
{
    while(scanf("%d%d",&n,&m)&& n && m)
    {

        for(int i = 1; i <= n; i++)  G[i].clear();
        for(int i = 1; i <= m; i++)
        {
            int u,v,dist;
            scanf("%d%d%d",&u,&v,&dist);
            edge e1,e2;
            e1.to = v, e1.cost = dist;
            e2.to = u, e2.cost = dist;
            G[u].push_back(e1);
            G[v].push_back(e2);
        }
        if(Spfa())
        printf("%d\n",d[n]);
    }
    return 0;
}
//Floyd
#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
typedef pair <int,int> P;
const int maxn = 100+10;
const int maxm = 1e4 + 100;
const int INF = 0x3f3f3f3f;
int n,m;
int mp[maxn][maxn];

void Floyd()
{
    for(int k = 1; k <= n; k++)
        for(int i = 1; i <= n; i++)
          for(int j = 1; j <= n; j++)
               mp[i][j] = min(mp[i][j],mp[i][k]+mp[k][j]);
}

int main()
{
    while(scanf("%d%d",&n,&m)&& n && m)
    {
        memset(mp,INF,sizeof(mp));
        int j = 1;
        for(int i = 1; i <= m; i++)
        {
            int u,v,dist;
            scanf("%d%d%d",&u,&v,&dist);
            mp[u][v] = dist;
            mp[v][u] = dist;
        }
        Floyd();
        printf("%d\n",mp[1][n]);
    }
    return 0;
}
//Bellman_Ford
#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
typedef pair <int,int> P;
const int maxn = 100+10;
const int maxm = 1e4 + 100;
const int INF = 0x3f3f3f3f;
int n,m;
int d[maxn];

struct edge {
  int from,to,cost;
};

edge es[maxm*2];

void Bellman_Ford()
{
    memset(d,INF,sizeof(d));
    d[1] = 0;
    for(int i = 1; i < n; i++)
    {
        for(int j = 1; j <= 2*m; j++)
        {
            edge e = es[j];
            if(d[e.from] < INF && d[e.to] > d[e.from] + e.cost)  d[e.to] = d[e.from] + e.cost;
        }
    }
}

int main()
{
    while(scanf("%d%d",&n,&m)&& n && m)
    {
        int j = 1;
        for(int i = 1; i <= m; i++)
        {
            int u,v,dist;
            scanf("%d%d%d",&u,&v,&dist);
            edge e1,e2;
            e1.from = u,e1.to = v,e1.cost = dist;
            e2.from = v,e2.to = u,e2.cost = dist;
            es[j++] = e1;
            es[j++] = e2;
        }
        Bellman_Ford();
        printf("%d\n",d[n]);
    }
}

3. HDU3790 最短路径问题

【题目描述】

给你n个点，m条无向边，每条边都有长度d和花费p，给你起点s终点t，要求输出起点到终点的最短距离及其花费，如果最短距离有多条路线，则输出花费最少的。

【输入】

输入n,m，点的编号是1~n,然后是m行，每行4个数 a,b,d,p，表示a和b之间有一条边，且其长度为d，花费为p。最后一行是两个数 s,t;起点s，终点。n和m为0时输入结束。
(1<n<=1000, 0<m<100000, s != t)

【输出】

输出 一行有两个数， 最短距离及其花费。

【样例输入】

3 2
1 2 5 6
2 3 4 5
1 3
0 0

【样例输出】

9 11

【解析】

最短距离 + 花费的问题，先优先满足距离，然后考虑花费，具体操作详见代码

【代码】

#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
const int INF = 0x3f3f3f3f;
const int maxn = 1e3+10;
const int maxm = 1e5+100;
int n,m,s,t;
struct P
{
    int u,d,c;
    P (int x, int y, int z) : u(x), d(y), c(z) {}
    bool operator >  (const P &a) const
    {
        if(this -> d != a.d)  return this->d > a.d;
        else                  return this->c > a.c;
    }
};

struct edge {
   int to,dist,cost;
   edge (int u,int d,int c) : to(u),dist(d),cost(c) {}
};

vector <edge> G[maxn];
void input()
{
    for(int i = 1; i <= n; i++) G[i].clear();
   for(int i = 0; i < m; i++)
   {
       int u,v,d,c;
       scanf("%d%d%d%d",&u,&v,&d,&c);
       G[u].push_back(edge(v,d,c));
       G[v].push_back(edge(u,d,c));
   }
   scanf("%d%d",&s,&t);
}

int d[maxn];
int c[maxn];

void dijkstra()
{
   priority_queue < P, vector <P> ,greater<P> > Q;
   Q.push(P(s,0,0));
   memset(d,INF,sizeof(d));
   memset(c,INF,sizeof(c));
   d[s] = 0;
   c[s] = 0;
   while(!Q.empty())
   {
       P temp = Q.top();
       Q.pop();
       int u = temp.u;
       int dist = temp.d;
       int cost = temp.c;

       if(d[u] < dist) continue;
       for(int i = 0; i < G[u].size(); i++)
       {
           edge e = G[u][i];
           if(d[e.to] > d[u] + e.dist ||(d[e.to] == d[u] + e.dist && c[e.to] > c[u] + e.cost))
           {
               d[e.to] = d[u] + e.dist;
               c[e.to] = c[u] + e.cost;
               Q.push(P(e.to,d[e.to],c[e.to]));
           }
       }
   }
   return ;
}

void output()
{
    printf("%d %d\n",d[t],c[t]);
}

int main()
{
  while(scanf("%d%d",&n,&m) && n && m)
  {
  input();
  dijkstra();
  output();
  }
  return 0;
}

4. HDU 2066 一个人的旅行

【题目描述】

虽然草儿是个路痴（就是在杭电待了一年多，居然还会在校园里迷路的人，汗),但是草儿仍然很喜欢旅行，因为在旅途中 会遇见很多人（白马王子，^0^），很多事，还能丰富自己的阅历，还可以看美丽的风景……草儿想去很多地方，她想要去东京铁塔看夜景，去威尼斯看电影，去阳明山上看海芋，去纽约纯粹看雪景，去巴黎喝咖啡写信，去北京探望孟姜女……眼看寒假就快到了，这么一大段时间，可不能浪费啊，一定要给自己好好的放个假，可是也不能荒废了训练啊，所以草儿决定在要在最短的时间去一个自己想去的地方！因为草儿的家在一个小镇上，没有火车经过，所以她只能去邻近的城市坐火车（好可怜啊）

【输入】

输入数据有多组，每组的第一行是三个整数T，S和D，表示有T条路，和草儿家相邻的城市的有S个，草儿想去的地方有D个；
接着有T行，每行有三个整数a，b，time,表示a,b城市之间的车程是time小时；(1=<(a,b)<=1000;a,b 之间可能有多条路)
接着的第T+1行有S个数，表示和草儿家相连的城市；
接着的第T+2行有D个数，表示草儿想去地方。

【输出】

输出草儿能去某个喜欢的城市的最短时间。

【样例输入】

6 2 3
1 3 5
1 4 7
2 8 12
3 8 4
4 9 12
9 10 2
1 2
8 9 10

【样例输出】

9 11

【解析】

此题看起来是多对多，但是实际上还是1对多的单源最短路问题，只需要把S集合的点都连向一个共同的起点S，权值为0，这样跑最短路模版即可。

【代码】

#include <bits/stdc++.h>

using namespace std;
typedef pair<int,int> P;
const int INF = 0x3f3f3f3f;
const int maxn = 1000+10;

int n = maxn-1,T,S,D;
int d[maxn];

struct edge
{
   int to,time;
   edge (int u,int c) : to(u),time(c) {}
};

vector <edge> G[maxn];

void input()
{
    for(int i = 0; i <= maxn; i++) G[i].clear();
    for(int i = 1; i <= T; i++)
    {
      int a,b,time;
      scanf("%d%d%d",&a,&b,&time);
      G[a].push_back(edge(b,time));
      G[b].push_back(edge(a,time));
    }
    for(int i = 1; i <= S; i++)
    {
        int s1;
        scanf("%d",&s1);
        G[0].push_back(edge(s1,0));
    }
    for(int i = 1; i <= D; i++)
    {
        int d1;
        scanf("%d",&d1);
        G[d1].push_back(edge(n,0));
    }
}

void dijkstra()
{
    priority_queue <P, vector<P>,greater<P> > Q;
    Q.push(P(0,0));
    memset(d,INF,sizeof(d));
    d[0] = 0;
    while(!Q.empty())
    {
        P x = Q.top();
        Q.pop();
        int u = x.second;
        int time = x.first;
        if(d[u] < time) continue;
        for(int i = 0; i < G[u].size(); i++)
        {
            edge e = G[u][i];
            if(d[e.to] > d[u] + e.time)
            {
                d[e.to] = d[u] + e.time;
                Q.push(P(d[e.to],e.to));
            }
        }
    }
    return ;
}

void output()
{
    printf("%d\n",d[n]);
}

int main()
{
    while(~scanf("%d%d%d",&T,&S,&D))
    {
    input();
    dijkstra();
    output();
    }
    return 0;
}

5. HDU2112 HDU Today

【题目描述】
经过锦囊相助，海东集团终于度过了危机，从此，HDU的发展就一直顺风顺水，到了2050年，集团已经相当规模了，据说进入了钱江肉丝经济开发区500强。这时候，XHD夫妇也退居了二线，并在风景秀美的诸暨市浬浦镇陶姚村买了个房子，开始安度晚年了。
这样住了一段时间，徐总对当地的交通还是不太了解。有时很郁闷，想去一个地方又不知道应该乘什么公交车，在什么地方转车，在什么地方下车（其实徐总自己有车，却一定要与民同乐，这就是徐总的性格）。
徐总经常会问蹩脚的英文问路：“Can you help me?”。看着他那迷茫而又无助的眼神，热心的你能帮帮他吗？
请帮助他用最短的时间到达目的地（假设每一路公交车都只在起点站和终点站停，而且随时都会开）。

【输入】

输入数据有多组，每组的第一行是公交车的总数N(0<=N<=10000)；
第二行有徐总的所在地start，他的目的地end；
接着有n行，每行有站名s，站名e，以及从s到e的时间整数t(0<t<100)(每个地名是一个长度不超过30的字符串)。
note：一组数据中地名数不会超过150个。
如果N==-1，表示输入结束。

【输出】

如果徐总能到达目的地，输出最短的时间；否则，输出“-1”。

【样例输入】

6
xiasha westlake
xiasha station 60
xiasha ShoppingCenterofHangZhou 30
station westlake 20
ShoppingCenterofHangZhou supermarket 10
xiasha supermarket 50
supermarket westlake 10
-1

【样例输出】

50

【解析】

模版题, 将地名字符串用map映射成一个整数，然后建图跑最短路即可。

【代码】

若n=0输入结束.

【输出】

对于每一个测试点如果存在能赚钱的情况则输出YES否则输出NO

【样例输入】

3
a
b
c
3
a 0.5 b
b 10.0 c
c 0.21 a

3
a
b
c
6
a 0.5 b
a 4.9 c
b 10.0 c
b 1.99 a
c 0.09 b
c 0.19 a

0

【样例输出】

Case 2: No

【解析】

如果可以套汇成功，则从起点到起点的汇率乘积>1, 因为是乘法，并且是求最大值，如果需要使用最短路算法，则需要取负对数。
因为起点有多个，所以需要用多源最短路算法Floyd。注意跑完Floyd后还需要双重循环来构造环。

【代码】

#include <bits/stdc++.h>

using namespace std;
typedef long long LL;

const int maxn = 40;

double G[maxn][maxn];
int n,m;
int T;
map <string,int> cnt;

void input()
{
    memset(G,0,sizeof(G));
    cnt.clear();
    int j = 0;
    for(int i = 0; i < n; i++)
    {
        string str;
        cin >> str;
        cnt[str] = j++;
    }
    scanf("%d",&m);
    for(int i = 0; i <m; i++)
    {
        string s1,s2;
        double c;
        cin >> s1;
        scanf("%lf",&c);
        cin >> s2;
        c = (-1) * log(c);
        G[cnt[s1]][cnt[s2]] = c;
    }

}


void Floyd()
{
    for(int k = 0; k < n; k++)
        for(int i = 0; i < n; i++)
          for(int j = 0; j < n; j++)
               G[i][j] = min(G[i][j],G[i][k]+G[k][j]);

    for(int i = 0; i < n; i++)
        for(int j = 0; j < n; j++)
           G[i][i] = min(G[i][i],G[i][j]+G[j][i]);
}

void output()
{
    for(int i = 0; i < n; i++)
    if(G[i][i] < -1e-8)  {printf("Case %d: Yes\n",T);return;}
    printf("Case %d: No\n",T);return ;
}

int main()
{
    T = 0;
    while(~scanf("%d",&n) && n)
    {
        T++;
        input();
        Floyd();
        output();
    }
    return 0;
}

【题目描述】

#include <bits/stdc++.h>

using namespace std;
typedef long long LL;

const int INF = 0x3f3f3f3f;
const int maxn = 100+10;
const double ept = 1e-5;

int n;
double k;

struct Point
{
    double x,y;
};

struct edge
{
    int to;
    double d;
    edge (int x = 0, double y = 0) : to(x),d(y){}
};

struct P
{
    int u;
    double d;
    int step;
    P (int x, double y, int z) : u(x),d(y),step(z) {}
    bool operator > (const P &a) const
    {
        if(fabs(this->d-a.d) < ept) return this->step < a.step;
        return this->d < a.d;
    }
};

double distance(Point a, Point b)
{
    double X = (a.x-b.x);
    double Y = (a.y-b.y);
    return sqrt(X*X+Y*Y);
}

Point S[maxn];

void input()
{
   for(int i = 1; i <= n; i++)
       scanf("%lf%lf",&S[i].x,&S[i].y);
}
vector <edge> G[maxn];

void build()
{
    for(int i = 0; i < maxn; i++) G[i].clear();
    for(int i = 1; i <=n; i++)
        for(int j = i+1; j <= n; j++)
        {
                double dis;
        dis = distance(S[i],S[j]);
        if(fabs(dis-k) < ept || k - dis > ept)
        {
            G[i].push_back(edge(j,dis));
                G[j].push_back(edge(i,dis));
        }
        }
    S[0].x = 0,S[0].y = 0;
    for(int i = 1; i <= n; i++)
    {
        double dis = distance(S[0],S[i]) - 7.5;
        if(fabs(dis-k) < ept || k - dis > ept)
         G[0].push_back(edge(i,dis));

        dis =50 -  max(fabs(S[i].x),fabs(S[i].y));
        if(fabs(dis-k) < ept || k - dis > ept)
             G[i].push_back(edge(n+1,dis));
    }
    double dis;
    dis = 42.5;
    if(fabs(dis - k) < ept || k - dis > ept)
        G[0].push_back(edge(n+1,dis));
}

double d[maxn];
int step[maxn];

void dijkstra()
{
   for(int i = 0; i < maxn; i++) d[i] = INF*1.0;
   priority_queue < P, vector <P>, greater<P> > Q;
   d[0] = 0;
   step[0] = 0;
   Q.push(P(0,0,0));
   while(!Q.empty())
   {
      P x = Q.top();
      Q.pop();
     // cout << x.u << endl;
      if(fabs(d[x.u] - x.d) > ept && d[x.u] < x.d) continue;
      for(int i = 0; i < G[x.u].size(); i++)
      {
          edge e = G[x.u][i];
      if(fabs(d[e.to]-(d[x.u]+e.d)) > ept && d[e.to] > d[x.u] + e.d)
      {
          d[e.to] = d[x.u] + e.d;
          step[e.to] = step[x.u] + 1;
          Q.push(P(e.to,d[e.to],step[e.to]));
          }
    }
    }
 }

void output()
{
  if(fabs(d[n+1]-INF) < ept) printf("can't be saved\n");
  else printf("%.2f %d\n",d[n+1],step[n+1]);
}

int main()
{

        build();

   }

9. HDU1317 XYZZY

【题目描述】
最近，有人发现了如何在Y-Crate游戏设备上运行开源软件。许多有进取心的设计师开发了Advent风格的游戏，可以在Y-Crate上进行部署。你的工作是测试一些设计，看看哪些是可赢的。
每场比赛由多达100个房间组成。其中一个房间是起点，其中一个房间是终点。每个房间的能量值介于-100和+100之间。对于这些房间，两两之间有单向门连接。

玩家在起点时有100点能量。她可以通过任何连接她所在房间的单向门到另一个房间。这个房间的能量值会增加到玩家的能量储蓄。这个过程将一直持续到她进入终点获胜或因能量耗尽而死亡（或者沮丧地退出）。在她的冒险中，玩家可以多次进入同一个房间，每次都能获得房间所提供的能量。
【输入】
输入包含多组数据。每组数据都以房间数n开始。房间的编号顺序是从1（起点）到n（终点）。接下来是有关n个房间的信息。每个房间的输入包含一行或多行，包含：

·房间i的能量值
·离开房间i的单向门数量
·离开房间i的单向门可以到达的房间列表

注：起点和终点房间的能量值均为0。当输入-1时，表明输入结束。
【输出】
在每组数据对应的一行输出中，如果玩家有可能获胜，则输出“winnable”，否则输出“hopeless”。
【样例输入】
5
0 1 2
-60 1 3
-60 1 4
20 1 5
0 0
5
0 1 2
20 1 3
-60 1 4
-60 1 5
0 0
5
0 1 2
21 1 3
-60 1 4
-60 1 5
0 0
5
0 1 2
20 2 1 3
-60 1 4
-60 1 5
0 0
-1
【样例输出】
hopeless
hopeless
winnable
winnable
【解析】

  • 需要注意，房间能量可以为负，门为单向，所以不能使用Dijkstra，因为虽然是房间为点，能量为点权，但是在加和的时候会影响到Dijkstra的贪心结果，使贪心策略不成立。
  • 注意起点和终点房间的能量值均为0。当输入-1时，表明输入结束。
  • 同时，因为需要判正环且需要让正环发挥作用，所以入队次数条件需要大于n-1，但是为了避免只有无关正环而导致的死循环，需要设立一个合适的阈值，在这里我设置的1e4，可以根据时间复杂度进行调整。

【代码】

#include <bits/stdc++.h>

using namespace std;
const int maxn = 100 + 10;

int n;
int enery[maxn];
vector <int> G[maxn];
int st;
int ed;
void input()
{
   st = ed = 0;
   for(int i = 0; i < maxn; i++) G[i].clear();
   for(int i = 1; i <= n; i++)
   {
       scanf("%d",&enery[i]);
       if(enery[i] == 0)
       {
           if(st == 0) st = i;
           else        ed = i;
       }
       int m;
       scanf("%d",&m);
       for(int j = 0; j < m; j++)
       {
           int v;
           scanf("%d",&v);
           G[i].push_back(v);
       }
   }
}

int d[maxn];
bool inq[maxn];
int num[maxn];

bool Spfa()
{
    memset(d,0,sizeof(d));
    memset(inq,false,sizeof(inq));
    memset(num,0,sizeof(num)); //记录入队次数
    queue <int> que;
    d[st] = 100;
    inq[st] = true;
    num[st]++;
    que.push(st);
    while(!que.empty())
    {
        int u = que.front();
        que.pop();
        inq[u] = false;
        for(int i = 0; i < G[u].size();i++)
        {
            int v = G[u][i];
            if(d[u] + enery[v] <= 0) continue;
            if(v == ed) return true;

            if(d[v] < d[u] + enery[v])
            {
               d[v] = d[u] + enery[v];
               if(!inq[v] && ++num[v] < 1e4)
               {
                   que.push(v);
                    inq[v] = true;
               }
            }
        }
    }
    return false;
}

int main()
{
  while(scanf("%d",&n) && n != -1)
  {
      input();
      if(Spfa())
        printf("winnable\n");
      else
        printf("hopeless\n");
  }
  return 0;
}

const int maxn = 1000 +10;

int n;
string str[maxn];
int value[maxn];
void input()
{
    for(int i = 0; i < n; i++)
        {
            scanf("%d",&value[i]);
            cin >> str[i];
        }
}

vector <int> G[maxn];

void build()
{
    for(int i = 0; i < n; i++)  G[i].clear();
    for(int i = 0; i < n; i++)
        for(int j = 0; j < n; j++)
    {
        if(i == j) continue;
        int s = str[i].size();
        if(str[i][s-4] == str[j][0] && str[i][s-3] == str[j][1] && str[i][s-2] == str[j][2] && str[i][s-1] == str[j][3])
                    {

                     G[i].push_back(j);
                  //   cout << i << " " <<j << endl;
                    }
    }
}

int d[maxn];

void dijkstra()
{
    memset(d,INF,sizeof(d));
    priority_queue < P, vector <P>, greater<P> > Q;
    Q.push(P(0,0));
    d[0] = 0;
    while(!Q.empty())
    {
        P x = Q.top();
        Q.pop();
        int u = x.second;
        if(d[u] < x.first) continue;
        for(int i = 0; i < G[u].size(); i++)
        {
            int v = G[u][i];
            if(d[v] > d[u] + value[u])
            {
                d[v] = d[u] + value[u];
                //cout << v << " " << d[v] << endl;
                Q.push(P(d[v],v));
            }
        }
    }

}

void output()
{
    if(d[n-1] < INF) printf("%d\n",d[n-1]);
    else             printf("-1\n");
}
int main()
{
    while(scanf("%d",&n) && n)
    {
        input();
        build();
        dijkstra();
        output();
    }
}

12. HDU2923 Einbahnstrasse

【题目描述】
Einbahnstrasse(德文单行道)是一条车辆只能朝一个方向行驶的街道。有单行道的一个原因是为了使通过拥挤地区的交通更加顺畅。这在市中心很有用，尤其是像开罗和大马士革这样的老城。精心的计划保证了你可以从任何地点出发到达任何地点。然而，司机必须仔细计划他们的路线，以避免延长他们的行程由于单行道。有经验的司机知道在任何两个地点之间都有多条路线。不仅如此，在相同的两个地点之间可能有多条道路。知道任何两个地点之间的最短路径是必须的!这一点在驾驶难以操纵的车辆(垃圾车、拖车等)时尤为重要。

你刚在一家汽车拖车公司开始一份新工作。公司的车库里停着几辆拖车。拖车将一辆坏车的前轮或后轮提起，以便将其直接拉回公司的车库。你会接到来自城市各个地方的电话，说有坏车需要拖走。这些车必须按照你接到电话的顺序被拖走。你的工作是建议拖车司机走最近的路，以便把所有坏掉的汽车送回公司的车库里。在一天结束时，你必须向管理部门报告卡车行驶的总距离。
【输入】
您的程序将在一个或多个测试用例上进行测试。每个测试用例的第一行指定三个数字(N、C和R)，由一个或多个空格分隔。这个城市有N个不同的地点，包括公司的车库。C是坏车的数量。R是城市道路的数量。注意，0< N < 100, 0<=C < 1000, R < 10000。第二行是由C + 1组成的单词，第一行是公司车库的位置，其余的是坏车的位置。一个位置是一个由10个或更少的字母组成的单词。字母大小写很重要。在第二行之后，将会有精确的R行，每一条都描述一条道路。道路描述使用以下三种格式之一:

A -v -> B
A <-v - B
A <-v -> B

A和B是两个不同位置的名称，v是表示道路长度的正整数(不超过1000)。第一种格式指定了从位置a到B的单行道，第二种指定了从B到a的单行道，而最后一种指定了它们之间的双向行道。A、“箭头”和B之间用一个或多个空格隔开。测试用例的末尾指定了一行三个0 (N、C和R)。

下面示例中的测试用例与图中的测试用例相同。

【输出】
对于每个测试用例，使用以下格式打印所走过的总距离:
k. V
其中k是测试用例编号(从1开始)，中间是一个空格，V是结果。
【样例输入】
4 2 5
NewTroy Midvale Metrodale
NewTroy <-20-> Midvale
Midvale –50-> Bakerline
NewTroy <-5– Bakerline
Metrodale <-30-> NewTroy
Metrodale –5-> Bakerline
0 0 0
【样例输出】
1. 80
【解析】

  • 使用map映射城市名，注意道路方向。
  • 去程建图跑单源最短路。
  • 因为有来有回，所以建图的时候可以建立一个原图的反向图，然后跑最短路，得出多对一的距离。

【代码】

#include <bits/stdc++.h>

using namespace std;
typedef pair <int,int> P;
typedef long long LL;
const int INF = 0x3f3f3f3f;
const int maxn = 100 +10;
const int maxr = 10000+10;
const int maxc = 1000+10;
int n,c,r;
int C[maxc];
int T = 0;
struct edge {
   int to,cost;
   edge (int x, int y) : to(x),cost(y) {}
};

map <string,int> cnt;

vector <edge> G[maxn];
vector <edge> H[maxn];
void input()
{
    for(int i = 0; i <= n; i++)  {G[i].clear();H[i].clear();}
    cnt.clear();
    int j = 0;
    string s;
    for(int i = 0; i <= c; i++)
    {
        cin >> s;
        if(cnt.count(s) == 0) cnt[s] = j++;
        C[i] = cnt[s];
    }

    for(int i = 0; i < r; i++)
    {
        string s1,s2,s3;
        cin >> s1 >> s2 >> s3;
        if(cnt.count(s1) == 0) cnt[s1] = j++;
        if(cnt.count(s3) == 0) cnt[s3] = j++;

        int u = cnt[s1];
        int v = cnt[s3];

        int len = s2.size();
        int sum = 0;
        int xp = 1;
        for(int i = len-1; i >= 0; i--)
        {
            if(s2[i] >= '0' && s2[i] <= '9')  {sum += (s2[i]-'0')*xp; xp *= 10;}
        }

        if(s2[0] == '<')     {G[v].push_back(edge(u,sum));H[u].push_back(edge(v,sum));}
        if(s2[len-1] == '>') {G[u].push_back(edge(v,sum));H[v].push_back(edge(u,sum));}

    }
}

int d[maxn];
int l[maxn];

void dijkstra()
{
    memset(d,INF,sizeof(d));
    memset(l,INF,sizeof(l));
    priority_queue <P, vector<P>, greater<P> > Q;
    d[0] = 0;
    Q.push(P(0,0));
    while(!Q.empty())
    {
        P x = Q.top();
        Q.pop();
        int u = x.second;
        if(d[u] < x.first)  continue;
        for(int i = 0; i < G[u].size(); i++)
            {
                edge e = G[u][i];
                if(d[e.to] > d[u] + e.cost)
                {
                    d[e.to] = d[u] + e.cost;
                    Q.push(P(d[e.to],e.to));
                }
            }
    }

    l[0] = 0;
    Q.push(P(0,0));
    while(!Q.empty())
    {
        P x = Q.top();
        Q.pop();
        int u = x.second;
        if(l[u] < x.first)  continue;
        for(int i = 0; i < H[u].size(); i++)
            {
                edge e = H[u][i];
                if(l[e.to] > l[u] + e.cost)
                {

                    l[e.to] = l[u] + e.cost;
                    Q.push(P(l[e.to],e.to));
                }
            }
    }
}

void output()
{
    int ans = 0;
    for(int i = 1; i <= c; i++)
        ans += (d[C[i]] + l[C[i]]);
    printf("%d. %d\n",T,ans);
}

int main()
{
    while(scanf("%d%d%d",&n,&c,&r) && (n || c || r))
    {
        T++;
        input();
        dijkstra();
        output();
    }
    return 0;
}

13. HDU3339 In Action

【题目描述】

自1945年美国曼哈顿计划小组(Manhattan Project team)引爆第一颗核弹以来，全球核武器数量激增。
现在，在FZU的一个名叫AekdyCoin的疯男孩拥有一些核武器，想要摧毁我们的世界。幸运的是，我们神秘的间谍网得到了他的计划。现在，我们需要阻止它。
但这项艰巨的任务显然不容易。首先，我们知道核武器的操作系统是由一些相互连接的电站组成的，它们构成了一个巨大而复杂的电网。每个电站都有其电力价值。要启动核武器，它必须消耗电网一半的电力。首先，我们需要使一半以上的电源失效。我们的坦克已经为在基地的行动做好了准备(ID为0)，我们必须在路上驾驶它们。至于发电站，只要我们的坦克停在那儿，我们就能控制它们。1单位距离花费1单位油。我们有足够的坦克来使用。
现在我们的指挥官想知道这次行动的最低石油成本。
【输入】
输入的第一行包含一个整数T，指定文件中测试用例的数量。
对于每种情况，第一行是整数n(1<= n<= 100)， m(1<= m<= 10000)，指定站点的数量(id为1、2、3…n)和站点之间的道路数量(双向)。
然后沿着m行，每一行是整数 st(0<= st<= n)， ed(0<= ed<= n)， dis(0<= dis<= 100)，指定开始点、结束点和之间的距离。
接n行，每一行为整数pow(1<= pow<= 100)，按ID顺序指定电站功率。
【输出】
这次行动的最低石油成本。
如果不存在，打印“impossible”(不带引号)。
【样例输入】
2
2 3
0 2 9
2 1 3
1 0 2
1
3
2 1
2 1 3
1
3
【样例输出】
5
impossible
【解析】

  • 最短路+动态规划
  • 先使用dijkstra求出基地到所有电站的距离，即每个电站的坦克耗油量。
  • 利用动态规划求出i单位油能控制的最大功率f[j]。
  • f[j] = max(f[j],f[j-d[i]]+w[i]); //同01背包问题，功率为价值，耗油量为体积。

【代码】

#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
typedef pair<int,int> P;
const int maxn = 100 + 10;
const int maxm = 1e4+10;
const int INF = 0x3f3f3f3f;
int n,m;
int w[maxn];
int W;
struct edge {
   int to,cost;
   edge(int x,int y) : to(x), cost(y) {}
};

vector <edge> G[maxn];

void input()
{
    for(int i = 0; i < maxn; i++) G[i].clear();
    scanf("%d%d",&n,&m);
    int st,ed,dis;
    for(int i = 0; i < m; i++)
    {
        scanf("%d%d%d",&st,&ed,&dis);
        G[st].push_back(edge(ed,dis));
        G[ed].push_back(edge(st,dis));
    }
    W = 0;
    for(int i = 1; i <= n; i++)
        {
            scanf("%d",&w[i]);
            W += w[i];
        }
}

int d[maxn];

void dijkstra()
{
    memset(d,INF,sizeof(d));
    d[0] = 0;
    priority_queue <P, vector<P>, greater<P> > Q;
    Q.push(P(0,0));
    while(!Q.empty())
    {
        P x = Q.top();
        Q.pop();
        int u = x.second;
        if(d[u] < x.first) continue;
        for(int i = 0; i < G[u].size(); i++)
        {
            edge e = G[u][i];
            if(d[e.to] > d[u] + e.cost)
            {
                d[e.to] = d[u] + e.cost;
                Q.push(P(d[e.to],e.to));
            }
        }
    }
}

int f[100010];

int solve()
{
    int lens = 0;
    for(int i = 1; i <= n; i++)  if(d[i] != INF) lens += d[i];
    memset(f,0,sizeof(f));
    for(int i = 1; i <= n; i++)
      {
          if(d[i] == INF) continue;
          for(int j = lens; j >=0; j--)
          if(j - d[i] >= 0)
          f[j] = max(f[j],f[j-d[i]]+w[i]);
      }
    for(int i = 1; i <= lens; i++)
        if(f[i]*2 > W) return i;
    return -1;
}

int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
      input();
      dijkstra();
      int ans = solve();
      if(ans == -1)     printf("impossible\n");
      else              printf("%d\n",ans);
    }
}
